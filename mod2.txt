function menuBotao(id, texto, funcao) {
    const botao = document.createElement('button');
    botao.id = id;
    botao.innerText = texto;
    botao.addEventListener('click', funcao);
    botao.style.color = 'orange';
    botao.style.backgroundColor = 'transparent';
    botao.style.borderColor = 'orange';
    botao.style.fontSize = '20px';
    overlay.appendChild(botao);   
}

// Funções de exemplo para os botões
function menu_Configuracoes() {
    console.log('Abrindo Configurações...');
}

function menu_Bloquear_Usuarios() {
    console.log('Bloqueando Usuários...');
}

// Criando os botões
menuBotao('menu_Configuracao', 'Configurações', menu_Configuracoes);
menuBotao('menu_Bloquear', 'Bloquear Usuários', menu_Bloquear_Usuarios);
